create table if not exists client (
    client_id int not null,
    client_name varchar(255) not null,
    primary key (client_id)
);

create table if not exists address (
    address_id int not null,
    address_number varchar(255) not null,
    address_line varchar(255) not null,
    primary key (address_id)
);

create table if not exists property_type (
    property_type_id int not null,
    property_type varchar(255) not null,
    primary key (property_type_id)
);

create table if not exists property (
    property_id int not null,
    address_id int not null,
    property_type_id int not null,
    primary key (property_id)
);

create table if not exists portfolio(
	portfolio_id int not null,
	client_id int not null,
	property_id int not null,
	primary key (portfolio_id)
);

create table if not exists tenant(
	tenant_id int not null,
	tenant_name varchar(255) not null,
	tenant_type_id int not null,
	primary key (tenant_id)
);

create table if not exists tenant_type (
    tenant_type_id int not null,
    tenant_type varchar(255) not null,
    primary key (tenant_type_id)
);

create table if not exists tenancy (
    tenancy_id int not null,
	tenant_id int not null,
	property_id int not null,
	start_date varchar(255) not null,
	end_date varchar(255) not null,
    monthly_rent int not null,
    primary key (tenancy_id)
);

create table if not exists part (
    part_id int not null,
    part_code varchar(255) not null,
    part_name varchar(255) not null,
    primary key (part_id)
);

create table if not exists staff (
   	staff_id int not null,
    staff_name varchar(255) not null,
    primary key (staff_id)
);

create table if not exists repair (
   	repair_id int not null,
    repair_details varchar(255) not null,
	property_id int not null,
    primary key (repair_id)
);

create table if not exists part_used (
   	part_used_id int not null,
    repair_id int not null,
	part_id int not null,
	quantity int not null,
    primary key (part_used_id)
);

create table if not exists repair_staff (
   	repair_staff_id int not null,
    repair_id int not null,
	staff_id int not null,
    primary key (repair_staff_id)
);
